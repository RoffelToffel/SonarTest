package com.acme.basic;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.InputStream;
import java.io.OutputStream;

public class ImproperPathnameLimitation {

	public static void main(String[] args) throws Exception {
		// The file that we want to upload
		File fileToUpload = new File("/path/to/file.exe");

		// Open a stream to the file
		InputStream inputStream = new FileInputStream(fileToUpload);

		// Save the file to the server without checking the file type
		saveFileToServer(inputStream);
	}

	public static void saveFileToServer(InputStream inputStream) throws Exception {
		// Create a buffer to read the file into
		byte[] buffer = new byte[4096];

		// Save the file to the server
		OutputStream outputStream = new FileOutputStream("/server/uploaded_file.exe");
		int bytesRead;
		while ((bytesRead = inputStream.read(buffer)) != -1) {
			outputStream.write(buffer, 0, bytesRead);

		}
	}
}
