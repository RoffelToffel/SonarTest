package com.acme.basic;

import java.util.Scanner;

public class Testing {

	public void test() {

		Scanner scan = new Scanner(System.in);
		// System.out.println("Enter n");
		int n = scan.nextInt();
		System.out.println(n);
		int a[][] = new int[n][n];
		int b = 0;
		int mid = n / 2 + 1;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (i + j == mid) {
					a[i][j] = n - b;
					b++;
				} else {
					a[i][j] = n;
				}

			}
		}
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				System.out.print(a[i][j]);
			}
			System.out.println();
		}

	}
}
